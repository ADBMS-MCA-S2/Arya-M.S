CREATE DATABASE moviee;
use moviee;
CREATE TABLE Movies(
    title VARCHAR(50),
    director VARCHAR(50),
    myear INT(50),
    rating FLOAT NOT NULL,
    PRIMARY KEY(title)
    );
CREATE TABLE Actors(
    actor VARCHAR(100),
    ayear INT NOT NULL,
    PRIMARY KEY(actor)
    );
CREATE TABLE Acts(
    actor VARCHAR(100),
    title VARCHAR(100),
    FOREIGN KEY(actor )REFERENCES Actors (actor) ON DELETE CASCADE,
    FOREIGN KEY(title) REFERENCES Movies (title) ON DELETE CASCADE
   );
CREATE TABLE Directors(
    director VARCHAR(100),
    dyear INT NOT NULL,
    PRIMARY KEY(director)
    );
